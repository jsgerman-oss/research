  ytick=data,

  yticklabel style={font=\footnotesize, text width=3.2cm, align=right},

\caption{Per-section classification applied by the conservative trim

Eq.\,\eqref{eq:decision}), and \emph{redundant} (collapsed to a
pointer).
Source: \texttt{data/aggregated/section-residency.csv}.}
