\section{Probability-weighted scoring derivation}
\label{app:probweight-derivation}

\tothink{Liu et al.\ \cite{liu2023geval} state the rule informally;
this appendix derives it as the expectation of the choice-score map
under the judge's implied posterior over the choice set. Properties:
(i) reduces to argmax in the zero-temperature limit; (ii) is strictly
proper on the choice set under log-loss; (iii) is monotonic in the
choice scores. The derivation also clarifies that the API need only
expose the top-$k$ log-probs --- the tail contributes
$\mathcal{O}(\epsilon)$ when truncated at the choice set.}

\section{Temperature-scaling fit}
\label{app:tempscale-fit}

\tothink{state the golden-section search algorithm on $T \in [0.1, 10]$
minimising negative log-likelihood on the labelled calibration set;
convergence in $\mathcal{O}(\log(1/\epsilon))$ iterations; per-(judge,
template) cache at \texttt{evals/\_calibration/<judge>/temperature.yaml}.
Cite Guo et al.\ \cite{guo2017calibration} for the canonical recipe.}

\section{Bias diagnostics}
\label{app:bias-diagnostics}

\tothink{state the three diagnostics formally. Position-bias flip-rate:
fraction of (A,B) vs (B,A) pairs whose verdict flipped, averaged over
the calibration set. Verbosity-bias correlation:
$\rho(\mathrm{length}(x), \score(x))$ over a held-out diagnostic set
where length is a confounder. Self-preference advantage:
$\mathbb{E}[\score | \text{candidate from same family as judge}] -
\mathbb{E}[\score | \text{candidate from cross-family}]$, computed
on a paired probe set where the candidate quality is held constant
across families. Cite Zheng et al.\ \cite{zheng2023judging},
Dubois et al.\ \cite{dubois2024lcalpaca}, and Panickssery et al.\
\cite{panickssery2024selfbias}.}
